SELECT first_name, DEPARTMENT,Salary
from (
     select first_name,DEPARTMENT, Salary,
            Row_Number() over(partition by DEPARTMENT order by Salary DESC) as ranking
     from worker
     ) as x
where ranking = 1 ORDER BY Salary DESC
